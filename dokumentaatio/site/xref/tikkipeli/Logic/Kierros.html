16      }

20      }
21      
22      /*
23      * Metodin tarkoitus on selventää vain valtin vaihtoa.

25      * erikseen ja yhden kierroksen aikana voi julistaa myös useamman valtin.
26      */
27      public void vaihdaValtti(Maa maa) {
28          valtti = maa;
