4   * KortinArvo on tehty enumiksi sillä kymppi on Tikkipelissä järjestyksessä
5   * toiseksi suurin kortti, Ässän jälkeen. Aina kun haetaan kortin arvoa
6   * ja selvitetään tikin voittanutta korttia ei näin ollen tarvitse erikseen
7   * asettaa lisäehtoa ottamaan huomioon onko kymppi pelattu tikkiin vai ei.
8   */
9   public enum KortinArvo {
10      k6("6", 6), k7("7", 7), k8("8", 8), k9("9", 9), k10("10", 14), kJ("J", 11), kQ("Q", 12), kK("K", 13), kA("A", 15);
11  
12      private final int arvo;
13      private final String arvoString;
14  
15      private KortinArvo(String blaa, int arvo) {
16          this.arvoString = blaa;
17          this.arvo = arvo;
18      }
19      
20      public int getKortinArvo() {
21          return this.arvo;
22      }
23      
24      @Override
25      public String toString() {
26          return arvoString;
27      }

29  }
